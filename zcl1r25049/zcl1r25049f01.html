FORM get_flight_data .

*-- Get airline master
  CLEAR gs_scarr.
  SELECT SINGLE carrid carrname currcode url
    INTO CORRESPONDING FIELDS OF gs_scarr
    FROM scarr
   WHERE carrid EQ pa_carr.

  IF gs_scarr IS INITIAL.
    MESSAGE s037 DISPLAY LIKE 'E'.
    STOP.
  ENDIF.

*-- Get flight schedule
  SELECT carrid connid cityfrom airpfrom cityto airpto
    INTO CORRESPONDING FIELDS OF TABLE gt_spfli
    FROM spfli
   WHERE carrid EQ pa_carr.

ENDFORM.

FORM display_screen .

  IF go_container IS INITIAL.

    CLEAR : gs_fcat, gt_fcat.
    PERFORM set_fcat USING :  'X' 'CARRID'   'SPFLI' 'C' ' ',
                              'X' 'CONNID'   'SPFLI' ' ' ' ',
                              ' ' 'CITYFROM' 'SPFLI' ' ' 'X',
                              ' ' 'AIRPFROM' 'SPFLI' ' ' 'X',
                              ' ' 'CITYTO'   'SPFLI' ' ' ' ',
                              ' ' 'AIRPTO'   'SPFLI' ' ' ' '.
    PERFORM set_layout.
    PERFORM create_obj.

    CALL METHOD go_alv_grid->set_table_for_first_display
      EXPORTING
        is_variant      = gs_variant
        i_save          = 'A'
        i_default       = 'X'
        is_layout       = gs_layout
      CHANGING
        it_outtab       = gt_spfli
        it_fieldcatalog = gt_fcat.

  ENDIF.

ENDFORM.

FORM set_fcat USING pv_key pv_field pv_table pv_just pv_emph.

  gs_fcat-key       = pv_key.
  gs_fcat-fieldname = pv_field.
  gs_fcat-ref_table = pv_table.
  gs_fcat-just      = pv_just.
  gs_fcat-emphasize = pv_emph.
  APPEND gs_fcat TO gt_fcat.
  CLEAR gs_fcat.

ENDFORM.
