FORM get_airline_master .

*-- Airline master
  CLEAR gt_scarr.
  SELECT carrid carrname currcode url
    INTO CORRESPONDING FIELDS OF TABLE gt_scarr
    FROM scarr
   WHERE carrid IN so_carr.

  IF gt_scarr IS INITIAL.
    MESSAGE s001 DISPLAY LIKE 'E'.
    STOP.
  ENDIF.

ENDFORM.

FORM display_screen .

  IF go_cont IS INITIAL.

    CLEAR : gt_tfcat, gt_bfcat.
    PERFORM set_tfcat USING : 'X' 'CARRID'   'SCARR' 'C' ' ',
                              ' ' 'CARRNAME' 'SCARR' ' ' ' ',
                              ' ' 'CURRCODE' 'SCARR' 'C' ' ',
                              ' ' 'URL'      'SCARR' ' ' 'X'.

    PERFORM set_bfcat USING : 'X' 'CARRID'   'SPFLI' 'C' ' ',
                              'X' 'CONNID'   'SPFLI' ' ' ' ',
                              ' ' 'CITYFROM' 'SPFLI' ' ' ' ',
                              ' ' 'CITYTO'   'SPFLI' ' ' 'X'.
    PERFORM set_layo.
    PERFORM create_obj.

    SET HANDLER : lcl_event_handler=>double_click FOR go_top_alv.

*-- # alv
    gs_variant-handle = 'ALV1'.
    CALL METHOD go_top_alv->set_table_for_first_display
      EXPORTING
        is_variant      = gs_variant
        i_save          = 'A'
        i_default       = 'X'
        is_layout       = gs_layo
      CHANGING
        it_outtab       = gt_scarr
        it_fieldcatalog = gt_tfcat.

*-- ## alv
    gs_variant-handle = 'ALV2'.
    CALL METHOD go_bottom_alv->set_table_for_first_display
      EXPORTING
        is_variant      = gs_variant
        i_save          = 'A'
        i_default       = 'X'
        is_layout       = gs_layo
      CHANGING
        it_outtab       = gt_spfli
        it_fieldcatalog = gt_bfcat.

  ENDIF.

ENDFORM.

FORM create_obj .

  CREATE OBJECT go_cont
    EXPORTING
      side      = go_cont->dock_at_left
      extension = 5000.

*-- #### : Split screen
  CREATE OBJECT go_split_cont
    EXPORTING
      parent  = go_cont
      rows    = 2  "2#
      columns = 1. "1#

*-- ## container# ##
  CALL METHOD go_split_cont->get_container
    EXPORTING
      row       = 1 "1#
      column    = 1 "1#
    RECEIVING
      container = go_top_cont.

  CALL METHOD go_split_cont->get_container
    EXPORTING
      row       = 2 "2#
      column    = 1 "1#
    RECEIVING
      container = go_bottom_cont.

*-- ALV# ##
  CREATE OBJECT go_top_alv
    EXPORTING
      i_parent = go_top_cont.

  CREATE OBJECT go_bottom_alv
    EXPORTING
      i_parent = go_bottom_cont.

ENDFORM.

FORM get_schedule_data .

*-- ## alv# ### ## #### ####
  SELECT a~carrid, connid, cityfrom, cityto
    FROM spfli AS a INNER JOIN @gt_scarr AS b
      ON a~carrid EQ b~carrid
   WHERE a~carrid IN @so_carr
     AND connid   IN @so_conn
   INTO CORRESPONDING FIELDS OF TABLE @gt_schedule.

ENDFORM.
