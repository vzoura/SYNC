FORM get_base_data .
  CLEAR : gt_body, gs_body.
  SELECT a~bukrs a~belnr a~gjahr waers wrbtr menge meins
    INTO CORRESPONDING FIELDS OF TABLE gt_body
    FROM bkpf AS a INNER JOIN bseg AS b
      ON a~bukrs EQ b~bukrs
     AND a~belnr EQ b~belnr
     AND a~gjahr EQ b~gjahr
   WHERE a~bukrs EQ pa_bukrs
     AND a~gjahr EQ pa_gjahr
    ORDER BY a~belnr.

  IF gt_body IS INITIAL.
    MESSAGE s005 DISPLAY LIKE 'E'.
    STOP.
  ENDIF.

ENDFORM.

FORM display_screen .
  DATA : ls_variant TYPE disvariant.

  IF go_container IS NOT BOUND.
    PERFORM create_object.
    PERFORM set_layout.
    CLEAR : gt_fcat, gs_fcat.
    PERFORM set_field_data USING : 'X' 'BUKRS' 'BKPF' 'C' ' ',
                                   'X' 'BELNR' 'BKPF' 'C' ' ',
                                   'X' 'GJAHR' 'BKPF' 'C' ' ',
                                   ' ' 'WRBTR' 'BSEG' 'C' ' ',
                                   ' ' 'WAERS' 'BKPF' 'C' 'X',
                                   ' ' 'MENGE' 'BSEG' 'C' ' ',
                                   ' ' 'MEINS' 'BSEG' 'C' 'X'.

    CALL METHOD go_alv_grid->set_table_for_first_display
      EXPORTING
        is_variant      = ls_variant
        i_save          = 'A'
        i_default       = 'X'
        is_layout       = gs_layout
      CHANGING
        it_outtab       = gt_body
        it_fieldcatalog = gt_fcat.
  ENDIF.

ENDFORM.

FORM set_field_data  USING pv_key pv_field pv_table pv_just pv_emph.
  gs_fcat-key       = pv_key.
  gs_fcat-fieldname = pv_field.
  gs_fcat-ref_table = pv_table.
  gs_fcat-just      = pv_just.
  gs_fcat-emphasize = pv_emph.

*************************************
*-- ##### ## #### ### ### ## #### ##
*-- ##### ## ## ### ### ## #### ##
*-- ## ### ### #### case### ##
*************************************
  CASE pv_field.
    WHEN 'WRBTR'.
      gs_fcat-cfieldname = 'WAERS'. "Currency Field Name
    WHEN 'MENGE'.
      gs_fcat-qfieldname = 'MEINS'. "Quantity Field Name
  ENDCASE.

  APPEND gs_fcat TO gt_fcat.
  CLEAR gs_fcat.

ENDFORM.

FORM create_object .

    EXPORTING
      side      = go_container->dock_at_left
      extension = 5000.

  CREATE OBJECT go_alv_grid
    EXPORTING
      i_parent = go_container.

ENDFORM.
