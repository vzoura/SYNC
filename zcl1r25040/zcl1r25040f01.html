FORM get_flight_schedule  USING    pv_carrid
                          CHANGING pv_subrc.

  CLEAR gt_spfli.
  SELECT carrid connid cityfrom cityto
    INTO CORRESPONDING FIELDS OF TABLE gt_spfli
    FROM spfli
   WHERE carrid EQ pv_carrid.

  IF gt_spfli IS INITIAL.
    pv_subrc = 'F'.
  ELSE.
    pv_subrc = 'S'.
  ENDIF.

ENDFORM.
