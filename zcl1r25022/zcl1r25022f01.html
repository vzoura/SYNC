FORM get_tax_data .

  "#### ####
  DATA : lv_cnt TYPE i.

  "Get Tax Code Data
  CLEAR : gt_t007s.
  SELECT mwskz text1
    INTO CORRESPONDING FIELDS OF TABLE gt_t007s
    FROM t007s
   WHERE kalsm EQ pa_kalsm
     AND mwskz IN so_mwskz.

  IF gt_t007s IS INITIAL.
    MESSAGE s005 DISPLAY LIKE 'E'.
    STOP.
  ENDIF.

  lv_cnt = lines( gt_t007s ).

  MESSAGE s003 WITH lv_cnt.

ENDFORM.
