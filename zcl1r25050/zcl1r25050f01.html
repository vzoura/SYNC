FORM set_init_data .

  pa_carr = 'LH'.

  "### #### select-option# ### ## ## ### # ##
  _ranges so_conn 'I' 'BT' '0400' ''.

ENDFORM.

FORM get_flight_data .

*-- Get airline master : Tables# spfli# #### ### ### SQL

  SELECT SINGLE *
    FROM scarr
   WHERE carrid EQ pa_carr.

*-- Get flight schedule : Tables# spfli# #### ### ### SQL

  SELECT SINGLE *
    FROM spfli
   WHERE carrid EQ pa_carr
     AND connid IN so_conn.

  IF spfli IS INITIAL.
    MESSAGE s037 DISPLAY LIKE 'E'.
    STOP.
  ENDIF.

*-- Get flight data
  SELECT carrid connid fldate price
         currency seatsmax seatsocc
    INTO CORRESPONDING FIELDS OF TABLE gt_sflight
    FROM sflight
   WHERE carrid EQ pa_carr
     AND connid IN so_conn.

ENDFORM.

FORM display_screen .

  IF go_cont IS INITIAL.

    CLEAR : gs_fcat, gt_fcat.
    PERFORM set_fcat USING : 'X' 'CARRID'   'SFLIGHT' 'C' ' ',
                             'X' 'CONNID'   'SFLIGHT' ' ' ' ',
                             'X' 'FLDATE'   'SFLIGHT' 'C' ' ',
                             ' ' 'PRICE'    'SFLIGHT' ' ' ' ',
                             ' ' 'CURRENCY' 'SFLIGHT' 'C' 'X',
                             ' ' 'SEATSMAX' 'SFLIGHT' ' ' ' ',
                             ' ' 'SEATSOCC' 'SFLIGHT' ' ' ' '.

    PERFORM set_layo.
    PERFORM create_obj.

    CALL METHOD go_alv->set_table_for_first_display
      EXPORTING
        is_variant      = gs_variant
        i_save          = 'A'
        i_default       = 'X'
        is_layout       = gs_layo
      CHANGING
        it_outtab       = gt_sflight
        it_fieldcatalog = gt_fcat.

  ENDIF.

ENDFORM.
*&---------------------------------------------------------------------*
*& Form set_fcat
*&---------------------------------------------------------------------*
*& text
*&---------------------------------------------------------------------*
*& -->  p1        text
*& <--  p2        text
*&---------------------------------------------------------------------*
FORM set_fcat USING pv_key pv_field pv_table pv_just pv_emph.

  gs_fcat-key       = pv_key.
  gs_fcat-fieldname = pv_field.
  gs_fcat-ref_table = pv_table.
  gs_fcat-just      = pv_just.
  gs_fcat-emphasize = pv_emph.

  CASE pv_field.
    WHEN 'PRICE'.
      gs_fcat-cfieldname = 'CURRENCY'.
  ENDCASE.

  APPEND gs_fcat TO gt_fcat.
  CLEAR gs_fcat.

ENDFORM.
